## 题意

You are given an array with all the numbers from 1 to N appearing exactly once, except for two number that is missing. How can you find the missing number in O(N) time and 0(1) space?

impl Solution {
    pub fn missing_two(nums: Vec<i32>) -> Vec<i32> {
        let nums = &nums;
        let n = nums.len() as i32 + 2;
        let mut key = 0;
        (1..=n).into_iter().for_each(|x| key ^= x);
        nums.into_iter().for_each(|x| key ^= x);
        let diff = key & -key;
        let mut one = 0;
        (1..=n).into_iter().for_each(|x| {
            if x & diff != 0 {
                one ^= x
            }
        });
        nums.into_iter().for_each(|x| {
            if x & diff != 0 {
                one ^= x
            }
        });
        vec![one, key ^ one]
    }
}

    println!("res:{:?}", Solution::missing_two(vec![1]));
}
